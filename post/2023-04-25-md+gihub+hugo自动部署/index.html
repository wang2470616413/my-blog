import argparse
import os
import re
import shutil
import subprocess


def fun():
    print(args)
    # 读取文件内容
    replaeTxt = ""
    with open(args.file, 'r',encoding="utf-8") as f:
        lines = f.readlines()
        for line in lines:
            # 对每一行进行处理
            pattern = r'\!\[.*\]\(.*\.png\)'
            match = re.match(pattern, line)
            # 输出匹配结果
            if (match and "/" not in line):
                print(line)
                if("\n" in line):
                    absFile = line.split("](")[-1][:-2]
                    line =line.split("](")[0]+"]("+ "/img" + "/" + line.split("\\")[-1]
                else:
                    absFile = line.split("](")[-1][:-1]
                    line = line.split("](")[0]+"]("+ "/img" + "/" + line.split("\\")[-1]

                print(absFile)
                shutil.copy(absFile, os.path.join(args.newFilePath,os.path.basename(absFile)))
            replaeTxt += line
            print(replaeTxt)
        with open(os.path.join(r"D:\Document\wordProjects\hugo\Sites\blog\content\post",args.file), "w",encoding="utf-8") as f:
            f.write(replaeTxt)
def cmdList(commandList):
    for cmd in commandList:
        result =subprocess.run(cmd['content'], stdout=subprocess.PIPE, shell=True,cwd=cmd['path'])
        print(result.stdout.decode('utf-8'))

if __name__ == '__main__':
    # 解析命令行参数
    parser = argparse.ArgumentParser()
    parser.add_argument('--file', type=str, help='file name',default="分析六级不过的原因以及接下来时间如何准备六级.md")
    parser.add_argument('--newFilePath', type=str, help='new text',default=r"D:\Document\wordProjects\hugo\Sites\blog\static\img")
    args = parser.parse_args()
    fun()
    print("请输入commit的内容")
    commitStr = str(input())
    commandList = [
        {'content': 'hugo', 'path': r'D:\Document\wordProjects\hugo\Sites\blog'},
        {'content': 'git add -A', 'path': r'D:\Document\wordProjects\hugo\Sites\blog\public'},
        {'content': 'git commit -m {}'.format(commitStr), 'path': r'D:\Document\wordProjects\hugo\Sites\blog\public'},
        {'content': 'git push', 'path': r'D:\Document\wordProjects\hugo\Sites\blog\public'}
    ]
    cmdList(commandList)

把这个脚本放在和md文件同级的目录下，然后执行

python .\test1.py --file '.\分析本次(2023-4-25)英语六级的成绩.md'
